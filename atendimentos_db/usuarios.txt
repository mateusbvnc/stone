-- ------------------------------------------------------------------------------------------------------------
-- criação de role de leitura
-- ------------------------------------------------------------------------------------------------------------

Create role reader;

Grant SELECT  on LastMile_schema.atendimentos to reader;

-- ------------------------------------------------------------------------------------------------------------
-- criação de role para escrita
-- ------------------------------------------------------------------------------------------------------------

Create role writer;

Grant INSERT, UPDATE, DELETE on LastMile_schema.atendimentos to writer; 

-- ------------------------------------------------------------------------------------------------------------
-- criação de usuario para leitura
-- ------------------------------------------------------------------------------------------------------------

Create user atendimentos_reader with password 'reader_teste';
Grant reader to atendimentos_reader;

-- ------------------------------------------------------------------------------------------------------------
-- criação de usuario para escrita
---------------------------------------------------------------------------------------------------------------

Create user atendimentos_writer with password 'writer_teste';
Grant writer to atendimentos_writer;
